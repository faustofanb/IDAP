#### 2.3.1 JWT 认证

**JwtTokenProvider.java**

```java
@Component
public class JwtTokenProvider {
    @Value("${jwt.secret}")
    private String secret;

    @Value("${jwt.expiration}")
    private long expiration;

    public String generateToken(UserDetails userDetails) {
        Map<String, Object> claims = new HashMap<>();
        claims.put("role", userDetails.getAuthorities().stream()
            .findFirst().map(GrantedAuthority::getAuthority).orElse("ROLE_USER"));

        return Jwts.builder()
            .setClaims(claims)
            .setSubject(userDetails.getUsername())
            .setIssuedAt(new Date())
            .setExpiration(new Date(System.currentTimeMillis() + expiration * 1000))
            .signWith(SignatureAlgorithm.HS512, secret)
            .compact();
    }

    public boolean validateToken(String token) {
        try {
            Jwts.parser().setSigningKey(secret).parseClaimsJws(token);
            return true;
        } catch (JwtException | IllegalArgumentException e) {
            return false;
        }
    }
}
```

```java
@RestController
@RequestMapping("/api/v1/query")
public class QueryController {

    @GetMapping(value = "/stream", produces = MediaType.TEXT_EVENT_STREAM_VALUE)
    public SseEmitter streamQuery(
        @RequestParam String question,
        @RequestParam String sessionId,
        @AuthenticationPrincipal UserDetails userDetails
    ) {
        SseEmitter emitter = new SseEmitter(60000L);

        executor.execute(() -> {
            try {
                // 1. 调用 Python AI 生成 SQL
                pythonAIClient.generateSqlStream(question, schema, (chunk) -> {
                    emitter.send(SseEmitter.event()
                        .name("sql_generation")
                        .data(chunk));
                });

                // 2. 执行 SQL
                QueryResult result = sqlExecutorService.executeSql(sql, userDetails.getUsername());
                emitter.send(SseEmitter.event()
                    .name("sql_result")
                    .data(result));

                // 3. 生成图表配置

                emitter.send(SseEmitter.event()
                    .name("chart")
                    .data(chart));

                // 4. 生成洞察
                pythonAIClient.generateInsightsStream(result, (chunk) -> {
                    emitter.send(SseEmitter.event()
                        .name("insights")
                        .data(chunk));
                });

                emitter.complete();
            } catch (Exception e) {
                emitter.completeWithError(e);
            }
        });

        return emitter;
```

### 3.3 核心功能设计

#### 3.3.1 LLM 服务

**llm_service.py**

```python
from langchain_openai import ChatOpenAI
from langchain_anthropic import ChatAnthropic

class LLMService:
    def __init__(self):
        self.models = {
            "gpt-4": ChatOpenAI(model="gpt-4-turbo-preview", temperature=0.7),
            "gpt-3.5": ChatOpenAI(model="gpt-3.5-turbo", temperature=0.7),
            "claude-3": ChatAnthropic(model="claude-3-sonnet-20240229")
        }

    async def generate(
        self,
        prompt: str,
        model: str = "gpt-4",
        stream: bool = False
    ):
        llm = self.models.get(model)
        if not llm:
            raise ValueError(f"Model {model} not supported")

        if stream:
            async for chunk in llm.astream(prompt):
                yield chunk.content
        else:
            response = await llm.ainvoke(prompt)
            return response.content
```

#### 3.3.2 NL2SQL Agent

**nl2sql/agent.py**

```python
from langchain.agents import create_sql_agent
from langchain.sql_database import SQLDatabase

class NL2SQLAgent:
    def __init__(self, llm_service: LLMService):
        self.llm = llm_service.models["gpt-4"]
        self.db = SQLDatabase.from_uri("postgresql://...")

    async def generate_sql(
        self,
        question: str,
        schema: dict
    ) -> dict:
        # 1. 检索相关表的 Schema
        relevant_tables = await self.schema_retriever.retrieve(question)

        # 2. 构建 Few-shot Prompt
        prompt = self._build_prompt(question, relevant_tables)

        # 3. 调用 LLM 生成 SQL
        sql = await self.llm.ainvoke(prompt)

        # 4. SQL 验证
        is_valid, error = self.sql_validator.validate(sql)

        return {
            "sql": sql,
            "confidence": 0.95,
            "tables_used": relevant_tables,
            "is_valid": is_valid,
            "error": error
        }

    def _build_prompt(self, question: str, tables: list) -> str:
        # 使用 Few-shot Prompt 模板
        return f"""
Given the following database schema:
{self._format_schema(tables)}

And these examples:
{self._get_few_shot_examples()}

Generate a SQL query to answer: {question}
"""
```

#### 3.3.3 RAG 检索器

**rag/retriever.py**

```python
from langchain.vectorstores import FAISS
from langchain.embeddings import OpenAIEmbeddings

class RAGRetriever:
    def __init__(self):
        self.embeddings = OpenAIEmbeddings(model="text-embedding-ada-002")
        self.vectorstore = FAISS.load_local("./data/faiss_index")

    async def retrieve(
        self,
        query: str,
        top_k: int = 5,
        similarity_threshold: float = 0.7
    ) -> list:
        # 1. 查询向量化
        query_embedding = await self.embeddings.aembed_query(query)

        # 2. 向量检索
        results = self.vectorstore.similarity_search_with_score(
            query,
            k=top_k * 2  # 多检索一些用于重排序
        )

        # 3. 过滤低相似度
        filtered = [
            (doc, score) for doc, score in results
            if score >= similarity_threshold
        ]

        # 4. 重排序
        reranked = await self.reranker.rerank(query, filtered)

        return reranked[:top_k]
```

#### 3.3.4 WebSocket 服务端

**websocket/handler.py**

```python
from fastapi import WebSocket

class WebSocketHandler:
    def __init__(self):
        self.active_connections: dict = {}

    async def connect(self, websocket: WebSocket, client_id: str):
        await websocket.accept()
        self.active_connections[client_id] = websocket

    async def handle_message(self, message: dict):
        request_id = message.get("id")
        method = message.get("method")
        params = message.get("params")

        if method == "nl2sql.generate":
            response = await self.nl2sql_agent.generate_sql(
                params["question"],
                params["schema"]
            )
            await self.send_response(request_id, response)

        elif method == "rag.query":
            async for chunk in self.rag_service.query_stream(params["question"]):
                await self.send_stream_data(request_id, chunk)
            await self.send_stream_end(request_id)

    async def send_response(self, request_id: str, result: dict):
        message = {
            "id": request_id,
            "type": "response",
            "result": result
        }
        await self.broadcast(message)
```

## 4. 数据流设计

### 4.1 NL2SQL 完整流程

```
1. 用户提问
   ↓
2. 前端 → Java Gateway (POST /api/v1/query)
   ↓
3. Java Gateway → Python AI (WebSocket: nl2sql.generate)
   ↓
4. Python AI 生成 SQL
   - LLM 生成
   - Schema 检索
   - Few-shot Prompt
   ↓
5. Python AI → Java Gateway (WebSocket: 返回 SQL)
   ↓
6. Java Gateway 执行 SQL
   - 白名单校验
   - 添加 LIMIT
   - HikariCP 执行
   ↓
7. Java Gateway 生成图表配置
   - 分析数据类型
   - 推荐图表类型
   - 生成 ECharts Config
   ↓
8. Java Gateway → 前端 (SSE: 流式返回)
   - SQL
   - 查询结果
   - 图表配置
   - 洞察分析
```

### 4.2 RAG 查询流程

```
1. 用户提问
   ↓
2. 前端 → Java Gateway
   ↓
3. Java Gateway → Python AI (WebSocket: rag.query)
   ↓
4. Python AI 向量检索
   - 问题向量化
   - FAISS/Milvus 检索
   - Top-K 召回
   - Reranking
   ↓
5. Python AI 生成答案
   - 检索结果 + 问题
   - LLM 生成答案
   - 保留引用来源
   ↓
6. Python AI → Java Gateway (WebSocket: 流式返回)
   ↓
7. Java Gateway → 前端 (SSE: 流式显示)
   - 答案内容
   - 引用来源
```
